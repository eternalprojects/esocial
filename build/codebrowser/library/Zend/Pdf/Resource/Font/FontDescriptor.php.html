  60.  60     /**

  66.  66      * available options and their descriptions.
  67.  67      *

  74.  74      * @param Zend_Pdf_Resource_Font $font Font
  75.  75      * @param Zend_Pdf_FileParser_Font_OpenType $fontParser Font parser object containing parsed TrueType file.

  91.  91          * attributes of the font. We will set as many of the bits as can be
  92.  92          * determined from the font parser.

  96.  96             $flags |= 1 << 0;
  97.  97         }

  99.  99             $flags |= 1 << 1;
  100. 100         }

  102. 102             $flags |= 1 << 2;
  103. 103         }
  104. 104         if ($fontParser->isScriptFont) {    // bit 4: Script
  105. 105             $flags |= 1 << 3;
  106. 106         }
  107. 107         if ($fontParser->isAdobeLatinSubset) {    // bit 6: Nonsymbolic
  108. 108             $flags |= 1 << 5;
  109. 109         }
  110. 110         if ($fontParser->isItalic) {    // bit 7: Italic
  111. 111             $flags |= 1 << 6;
  112. 112         }
  113. 113         // bits 17-19: AllCap, SmallCap, ForceBold; not available
  114. 114         $fontDescriptor->Flags = new Zend_Pdf_Element_Numeric($flags);
  115. 115 
  116. 116         $fontBBox = array(new Zend_Pdf_Element_Numeric($font->toEmSpace($fontParser->xMin)),
  117. 117                           new Zend_Pdf_Element_Numeric($font->toEmSpace($fontParser->yMin)),
  118. 118                           new Zend_Pdf_Element_Numeric($font->toEmSpace($fontParser->xMax)),
  119. 119                           new Zend_Pdf_Element_Numeric($font->toEmSpace($fontParser->yMax)));

  121. 121 
  122. 122         $fontDescriptor->ItalicAngle  = new Zend_Pdf_Element_Numeric($fontParser->italicAngle);
  123. 123 
  124. 124         $fontDescriptor->Ascent       = new Zend_Pdf_Element_Numeric($font->toEmSpace($fontParser->ascent));

  126. 126 
  127. 127         $fontDescriptor->CapHeight    = new Zend_Pdf_Element_Numeric($fontParser->capitalHeight);
  128. 128         /**

  130. 130          * file. For now, record zero which is interpreted as 'unknown'.

  133. 133         $fontDescriptor->StemV        = new Zend_Pdf_Element_Numeric(0);
  134. 134 
  135. 135         $fontDescriptor->MissingWidth = new Zend_Pdf_Element_Numeric($fontParser->glyphWidths[0]);
  136. 136 
  137. 137         /* Set up font embedding. This is where the actual font program itself
  138. 138          * is embedded within the PDF document.
  139. 139          *
  140. 140          * Note that it is not requried that fonts be embedded within the PDF
  141. 141          * document to use them. If the recipient of the PDF has the font
  142. 142          * installed on their computer, they will see the correct fonts in the
  143. 143          * document. If they don't, the PDF viewer will substitute or synthesize
  144. 144          * a replacement.
  145. 145          *
  146. 146          * There are several guidelines for font embedding:
  147. 147          *
  148. 148          * First, the developer might specifically request not to embed the font.
  149. 149          */
  150. 150         if (!($embeddingOptions & Zend_Pdf_Font::EMBED_DONT_EMBED)) {
  151. 151 
  152. 152             /* Second, the font author may have set copyright bits that prohibit
  153. 153              * the font program from being embedded. Yes this is controversial,
  154. 154              * but it's the rules:
  155. 155              *   http://partners.adobe.com/public/developer/en/acrobat/sdk/FontPolicies.pdf
  156. 156              *
  157. 157              * To keep the developer in the loop, and to prevent surprising bug
  158. 158              * reports of "your PDF doesn't have the right fonts," throw an

  161. 161             if (! $fontParser->isEmbeddable) {

  165. 165                 if (!($embeddingOptions & Zend_Pdf_Font::EMBED_SUPPRESS_EMBED_EXCEPTION)) {
  166. 166                     $message = 'This font cannot be embedded in the PDF document. If you would like to use '
  167. 167                              . 'it anyway, you must pass Zend_Pdf_Font::EMBED_SUPPRESS_EMBED_EXCEPTION '
  168. 168                              . 'in the $options parameter of the font constructor.';
  169. 169                     require_once 'Zend/Pdf/Exception.php';
  170. 170                     throw new Zend_Pdf_Exception($message, Zend_Pdf_Exception::FONT_CANT_BE_EMBEDDED);
  171. 171                 }
  172. 172 
  173. 173             } else {
  174. 174                 /* Otherwise, the default behavior is to embed all custom fonts.
  175. 175                  */
  176. 176                 /* This section will change soon to a stream object data
  177. 177                  * provider model so that we don't have to keep a copy of the
  178. 178                  * entire font in memory.

  183. 183                 $fontFile = $fontParser->getDataSource()->readAllBytes();

  185. 185                 $fontFileObject->dictionary->Length1 = new Zend_Pdf_Element_Numeric(strlen($fontFile));
  186. 186                 if (!($embeddingOptions & Zend_Pdf_Font::EMBED_DONT_COMPRESS)) {
  187. 187                     /* Compress the font file using Flate. This generally cuts file
  188. 188                      * sizes by about half!
  189. 189                      */
  190. 190                     $fontFileObject->dictionary->Filter = new Zend_Pdf_Element_Name('FlateDecode');
  191. 191                 }
  192. 192                 if ($fontParser instanceof Zend_Pdf_FileParser_Font_OpenType_Type1 /* not implemented now */) {
  193. 193                     $fontDescriptor->FontFile  = $fontFileObject;
  194. 194                 } else if ($fontParser instanceof Zend_Pdf_FileParser_Font_OpenType_TrueType) {
  195. 195                     $fontDescriptor->FontFile2 = $fontFileObject;
  196. 196                 } else {
  197. 197                     $fontDescriptor->FontFile3 = $fontFileObject;
  198. 198                 }
  199. 199             }
  200. 200         }
  201. 201 
  202. 202         return $fontDescriptor;
  203. 203     }
