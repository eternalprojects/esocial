  101. 101 
  102. 102         $compression = ord(fread($imageFile, 1));
  103. 103         $prefilter = ord(fread($imageFile,1));

  106. 106             require_once 'Zend/Pdf/Exception.php';
  107. 107             throw new Zend_Pdf_Exception( "Only non-interlaced images are currently supported." );

  119. 119         fseek($imageFile, 4, SEEK_CUR); //4 Byte Ending Sequence

  122. 122         /*

  124. 124          * followed by the chunk signature, a four byte code. IDAT and IEND are manditory in any PNG.
  125. 125          */

  128. 128             if ($chunkLengthBytes === false) {
  129. 129                 require_once 'Zend/Pdf/Exception.php';
  130. 130                 throw new Zend_Pdf_Exception('Error ocuured while image file reading.');
  131. 131             }

  134. 134             $chunkLength    = $chunkLengthtmp['i'];

  138. 138                     /*
  139. 139                      * Reads the actual image data from the PNG file. Since we know at this point that the compression
  140. 140                      * strategy is the default strategy, we also know that this data is Zip compressed. We will either copy
  141. 141                      * the data directly to the PDF and provide the correct FlateDecode predictor, or decompress the data
  142. 142                      * decode the filters and output the data as a raw pixel map.
  143. 143                      */
  144. 144                     $imageData .= fread($imageFile, $chunkLength);
  145. 145                     fseek($imageFile, 4, SEEK_CUR);
  146. 146                     break;

  148. 148                 case 'PLTE': //Palette
  149. 149                     $paletteData = fread($imageFile, $chunkLength);
  150. 150                     fseek($imageFile, 4, SEEK_CUR);
  151. 151                     break;

  153. 153                 case 'tRNS': //Basic (non-alpha channel) transparency.
  154. 154                     $trnsData = fread($imageFile, $chunkLength);
  155. 155                     switch ($color) {
  156. 156                         case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_GRAY:
  157. 157                             $baseColor = ord(substr($trnsData, 1, 1));
  158. 158                             $transparencyData = array(new Zend_Pdf_Element_Numeric($baseColor),
  159. 159                                                       new Zend_Pdf_Element_Numeric($baseColor));
  160. 160                             break;
  161. 161 
  162. 162                         case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_RGB:
  163. 163                             $red = ord(substr($trnsData,1,1));
  164. 164                             $green = ord(substr($trnsData,3,1));
  165. 165                             $blue = ord(substr($trnsData,5,1));
  166. 166                             $transparencyData = array(new Zend_Pdf_Element_Numeric($red),
  167. 167                                                       new Zend_Pdf_Element_Numeric($red),
  168. 168                                                       new Zend_Pdf_Element_Numeric($green),
  169. 169                                                       new Zend_Pdf_Element_Numeric($green),
  170. 170                                                       new Zend_Pdf_Element_Numeric($blue),
  171. 171                                                       new Zend_Pdf_Element_Numeric($blue));
  172. 172                             break;
  173. 173 
  174. 174                         case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_INDEXED:
  175. 175                             //Find the first transparent color in the index, we will mask that. (This is a bit of a hack. This should be a SMask and mask all entries values).
  176. 176                             if(($trnsIdx = strpos($trnsData, "\0")) !== false) {
  177. 177                                 $transparencyData = array(new Zend_Pdf_Element_Numeric($trnsIdx),
  178. 178                                                           new Zend_Pdf_Element_Numeric($trnsIdx));
  179. 179                             }
  180. 180                             break;
  181. 181 
  182. 182                         case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_GRAY_ALPHA:
  183. 183                             // Fall through to the next case
  184. 184 
  185. 185                         case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_RGB_ALPHA:
  186. 186                             require_once 'Zend/Pdf/Exception.php';
  187. 187                             throw new Zend_Pdf_Exception( "tRNS chunk illegal for Alpha Channel Images" );
  188. 188                             break;
  189. 189                     }
  190. 190                     fseek($imageFile, 4, SEEK_CUR); //4 Byte Ending Sequence
  191. 191                     break;
  192. 192 
  193. 193                 case 'IEND';
  194. 194                     break 2; //End the loop too
  195. 195 
  196. 196                 default:
  197. 197                     fseek($imageFile, $chunkLength + 4, SEEK_CUR); //Skip the section
  198. 198                     break;
  199. 199             }
  200. 200         }
  201. 201         fclose($imageFile);
  202. 202 
  203. 203         $compressed = true;
  204. 204         $imageDataTmp = '';
  205. 205         $smaskData = '';
  206. 206         switch ($color) {
  207. 207             case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_RGB:
  208. 208                 $colorSpace = new Zend_Pdf_Element_Name('DeviceRGB');
  209. 209                 break;
  210. 210 
  211. 211             case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_GRAY:
  212. 212                 $colorSpace = new Zend_Pdf_Element_Name('DeviceGray');
  213. 213                 break;
  214. 214 
  215. 215             case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_INDEXED:
  216. 216                 if(empty($paletteData)) {
  217. 217                     require_once 'Zend/Pdf/Exception.php';
  218. 218                     throw new Zend_Pdf_Exception( "PNG Corruption: No palette data read for indexed type PNG." );
  219. 219                 }

  221. 221                 $colorSpace->items[] = new Zend_Pdf_Element_Name('Indexed');
  222. 222                 $colorSpace->items[] = new Zend_Pdf_Element_Name('DeviceRGB');
  223. 223                 $colorSpace->items[] = new Zend_Pdf_Element_Numeric((strlen($paletteData)/3-1));
  224. 224                 $paletteObject = $this->_objectFactory->newObject(new Zend_Pdf_Element_String_Binary($paletteData));

  226. 226                 break;
  227. 227 
  228. 228             case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_GRAY_ALPHA:

  231. 231                  * the other will contain the Gray transparency overlay data. The former will become the object data and the latter
  232. 232                  * will become the Shadow Mask (SMask).
  233. 233                  */
  234. 234                 if($bits > 8) {
  235. 235                     require_once 'Zend/Pdf/Exception.php';
  236. 236                     throw new Zend_Pdf_Exception("Alpha PNGs with bit depth > 8 are not yet supported");
  237. 237                 }
  238. 238 
  239. 239                 $colorSpace = new Zend_Pdf_Element_Name('DeviceGray');
  240. 240 
  241. 241                 require_once 'Zend/Pdf/ElementFactory.php';
  242. 242                 $decodingObjFactory = Zend_Pdf_ElementFactory::createFactory(1);
  243. 243                 $decodingStream = $decodingObjFactory->newStreamObject($imageData);
  244. 244                 $decodingStream->dictionary->Filter      = new Zend_Pdf_Element_Name('FlateDecode');
  245. 245                 $decodingStream->dictionary->DecodeParms = new Zend_Pdf_Element_Dictionary();

  247. 247                 $decodingStream->dictionary->DecodeParms->Columns          = new Zend_Pdf_Element_Numeric($width);
  248. 248                 $decodingStream->dictionary->DecodeParms->Colors           = new Zend_Pdf_Element_Numeric(2);   //GreyAlpha
  249. 249                 $decodingStream->dictionary->DecodeParms->BitsPerComponent = new Zend_Pdf_Element_Numeric($bits);
  250. 250                 $decodingStream->skipFilters();
  251. 251 
  252. 252                 $pngDataRawDecoded = $decodingStream->value;
  253. 253 
  254. 254                 //Iterate every pixel and copy out gray data and alpha channel (this will be slow)

  256. 256                     $imageDataTmp .= $pngDataRawDecoded[($pixel*2)];
  257. 257                     $smaskData .= $pngDataRawDecoded[($pixel*2)+1];
  258. 258                 }
  259. 259                 $compressed = false;
  260. 260                 $imageData  = $imageDataTmp; //Overwrite image data with the gray channel without alpha
  261. 261                 break;
  262. 262 
  263. 263             case Zend_Pdf_Resource_Image_Png::PNG_CHANNEL_RGB_ALPHA:

  265. 265                  * To decode PNG's with alpha data we must create two images from one. One image will contain the RGB data
  266. 266                  * the other will contain the Gray transparency overlay data. The former will become the object data and the latter
  267. 267                  * will become the Shadow Mask (SMask).
  268. 268                  */

  270. 270                     require_once 'Zend/Pdf/Exception.php';
  271. 271                     throw new Zend_Pdf_Exception("Alpha PNGs with bit depth > 8 are not yet supported");
  272. 272                 }
  273. 273 
  274. 274                 $colorSpace = new Zend_Pdf_Element_Name('DeviceRGB');
  275. 275 
  276. 276                 require_once 'Zend/Pdf/ElementFactory.php';
  277. 277                 $decodingObjFactory = Zend_Pdf_ElementFactory::createFactory(1);
  278. 278                 $decodingStream = $decodingObjFactory->newStreamObject($imageData);
  279. 279                 $decodingStream->dictionary->Filter      = new Zend_Pdf_Element_Name('FlateDecode');
  280. 280                 $decodingStream->dictionary->DecodeParms = new Zend_Pdf_Element_Dictionary();
  281. 281                 $decodingStream->dictionary->DecodeParms->Predictor        = new Zend_Pdf_Element_Numeric(15);
  282. 282                 $decodingStream->dictionary->DecodeParms->Columns          = new Zend_Pdf_Element_Numeric($width);
  283. 283                 $decodingStream->dictionary->DecodeParms->Colors           = new Zend_Pdf_Element_Numeric(4);   //RGBA
  284. 284                 $decodingStream->dictionary->DecodeParms->BitsPerComponent = new Zend_Pdf_Element_Numeric($bits);
  285. 285                 $decodingStream->skipFilters();
  286. 286 
  287. 287                 $pngDataRawDecoded = $decodingStream->value;
  288. 288 
  289. 289                 //Iterate every pixel and copy out rgb data and alpha channel (this will be slow)

  291. 291                     $imageDataTmp .= $pngDataRawDecoded[($pixel*4)+0] . $pngDataRawDecoded[($pixel*4)+1] . $pngDataRawDecoded[($pixel*4)+2];
  292. 292                     $smaskData .= $pngDataRawDecoded[($pixel*4)+3];
  293. 293                 }
  294. 294 
  295. 295                 $compressed = false;
  296. 296                 $imageData  = $imageDataTmp; //Overwrite image data with the RGB channel without alpha
  297. 297                 break;
  298. 298 
  299. 299             default:
  300. 300                 require_once 'Zend/Pdf/Exception.php';
  301. 301                 throw new Zend_Pdf_Exception( "PNG Corruption: Invalid color space." );
  302. 302         }
  303. 303 
  304. 304         if(empty($imageData)) {
  305. 305             require_once 'Zend/Pdf/Exception.php';
  306. 306             throw new Zend_Pdf_Exception( "Corrupt PNG Image. Mandatory IDAT chunk not found." );
  307. 307         }
  308. 308 
  309. 309         $imageDictionary = $this->_resource->dictionary;
  310. 310         if(!empty($smaskData)) {
  311. 311             /*
  312. 312              * Includes the Alpha transparency data as a Gray Image, then assigns the image as the Shadow Mask for the main image data.
  313. 313              */
  314. 314             $smaskStream = $this->_objectFactory->newStreamObject($smaskData);
  315. 315             $smaskStream->dictionary->Type             = new Zend_Pdf_Element_Name('XObject');
  316. 316             $smaskStream->dictionary->Subtype          = new Zend_Pdf_Element_Name('Image');
  317. 317             $smaskStream->dictionary->Width            = new Zend_Pdf_Element_Numeric($width);
  318. 318             $smaskStream->dictionary->Height           = new Zend_Pdf_Element_Numeric($height);
  319. 319             $smaskStream->dictionary->ColorSpace       = new Zend_Pdf_Element_Name('DeviceGray');
  320. 320             $smaskStream->dictionary->BitsPerComponent = new Zend_Pdf_Element_Numeric($bits);
  321. 321             $imageDictionary->SMask = $smaskStream;
  322. 322 
  323. 323             // Encode stream with FlateDecode filter
  324. 324             $smaskStreamDecodeParms = array();
  325. 325             $smaskStreamDecodeParms['Predictor']        = new Zend_Pdf_Element_Numeric(15);
  326. 326             $smaskStreamDecodeParms['Columns']          = new Zend_Pdf_Element_Numeric($width);
  327. 327             $smaskStreamDecodeParms['Colors']           = new Zend_Pdf_Element_Numeric(1);
  328. 328             $smaskStreamDecodeParms['BitsPerComponent'] = new Zend_Pdf_Element_Numeric(8);
  329. 329             $smaskStream->dictionary->DecodeParms  = new Zend_Pdf_Element_Dictionary($smaskStreamDecodeParms);
  330. 330             $smaskStream->dictionary->Filter       = new Zend_Pdf_Element_Name('FlateDecode');
  331. 331         }
  332. 332 
  333. 333         if(!empty($transparencyData)) {
  334. 334             //This is experimental and not properly tested.
  335. 335             $imageDictionary->Mask = new Zend_Pdf_Element_Array($transparencyData);
  336. 336         }
