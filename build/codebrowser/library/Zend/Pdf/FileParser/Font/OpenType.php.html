  27.   27  * Abstract base class for OpenType font file parsers.
  28.   28  *

  141.  141     /**
  142.  142      * Parses the OpenType table directory.
  143.  143      *
  144.  144      * The table directory contains the identifier, checksum, byte offset, and
  145.  145      * length of each of the information tables housed in the font file.
  146.  146      *
  147.  147      * @throws Zend_Pdf_Exception
  148.  148      */
  149.  149     protected function _parseTableDirectory()
  150.  150     {
  151.  151         $this->moveToOffset(4);
  152.  152 
  153.  153         $tableCount = $this->readUInt(2);
  154.  154         $this->_debugLog('%d tables', $tableCount);
  155.  155 
  156.  156         /* Sanity check, in case we're not actually reading a OpenType file and
  157.  157          * the first four bytes coincidentally matched an OpenType signature in
  158.  158          * screen() above.
  159.  159          *
  160.  160          * There are at minimum 7 required tables: cmap, head, hhea, hmtx, maxp,
  161.  161          * name, and post. In the current OpenType standard, only 32 table types
  162.  162          * are defined, so use 50 as a practical limit.
  163.  163          */
  164.  164         if (($tableCount < 7) || ($tableCount > 50)) {
  165.  165             require_once 'Zend/Pdf/Exception.php';
  166.  166             throw new Zend_Pdf_Exception('Table count not within expected range',
  167.  167                                          Zend_Pdf_Exception::BAD_TABLE_COUNT);
  168.  168         }
  169.  169 
  170.  170         /* Skip the next 6 bytes, which contain values to aid a binary search.
  171.  171          */
  172.  172         $this->skipBytes(6);
  173.  173 
  174.  174         /* The directory contains four values: the name of the table, checksum,
  175.  175          * offset to the table from the beginning of the font, and actual data
  176.  176          * length of the table.
  177.  177          */
  178.  178         for ($tableIndex = 0; $tableIndex < $tableCount; $tableIndex++) {
  179.  179             $tableName = $this->readBytes(4);
  180.  180 
  181.  181             /* We ignore the checksum here for two reasons: First, the PDF viewer
  182.  182              * will do this later anyway; Second, calculating the checksum would
  183.  183              * require unsigned integers, which PHP does not currently provide.
  184.  184              * We may revisit this in the future.
  185.  185              */
  186.  186             $this->skipBytes(4);
  187.  187 
  188.  188             $tableOffset = $this->readUInt(4);
  189.  189             $tableLength = $this->readUInt(4);
  190.  190             $this->_debugLog('%s offset: 0x%x; length: %d', $tableName, $tableOffset, $tableLength);
  191.  191 
  192.  192             /* Sanity checks for offset and length values.
  193.  193              */
  194.  194             $fileSize = $this->_dataSource->getSize();
  195.  195             if (($tableOffset < 0) || ($tableOffset > $fileSize)) {
  196.  196                 require_once 'Zend/Pdf/Exception.php';
  197.  197                 throw new Zend_Pdf_Exception("Table offset ($tableOffset) not within expected range",
  198.  198                                              Zend_Pdf_Exception::INDEX_OUT_OF_RANGE);
  199.  199             }
  200.  200             if (($tableLength < 0) || (($tableOffset + $tableLength) > $fileSize)) {
  201.  201                 require_once 'Zend/Pdf/Exception.php';
  202.  202                 throw new Zend_Pdf_Exception("Table length ($tableLength) not within expected range",
  203.  203                                              Zend_Pdf_Exception::INDEX_OUT_OF_RANGE);
  204.  204             }
  205.  205 
  206.  206             $this->_tableDirectory[$tableName]['offset'] = $tableOffset;
  207.  207             $this->_tableDirectory[$tableName]['length'] = $tableLength;
  208.  208         }
  209.  209     }
  210.  210

  212.  212     /**
  213.  213      * Parses the OpenType head (Font Header) table.
  214.  214      *
  215.  215      * The head table contains global information about the font such as the
  216.  216      * revision number and global metrics.
  217.  217      *
  218.  218      * @throws Zend_Pdf_Exception
  219.  219      */
  220.  220     protected function _parseHeadTable()
  221.  221     {
  222.  222         $this->_jumpToTable('head');
  223.  223 
  224.  224         /* We can read any version 1 table.
  225.  225          */
  226.  226         $tableVersion = $this->_readTableVersion(1, 1);
  227.  227 
  228.  228         /* Skip the font revision number and checksum adjustment.
  229.  229          */
  230.  230         $this->skipBytes(8);
  231.  231 
  232.  232         $magicNumber = $this->readUInt(4);
  233.  233         if ($magicNumber != 0x5f0f3cf5) {
  234.  234             require_once 'Zend/Pdf/Exception.php';
  235.  235             throw new Zend_Pdf_Exception('Wrong magic number. Expected: 0x5f0f3cf5; actual: '
  236.  236                                        . sprintf('%x', $magicNumber),
  237.  237                                          Zend_Pdf_Exception::BAD_MAGIC_NUMBER);
  238.  238         }
  239.  239 
  240.  240         /* Most of the flags we ignore, but there are a few values that are
  241.  241          * useful for our layout routines.
  242.  242          */
  243.  243         $flags = $this->readUInt(2);
  244.  244         $this->baselineAtZero    = $this->isBitSet(0, $flags);
  245.  245         $this->useIntegerScaling = $this->isBitSet(3, $flags);
  246.  246 
  247.  247         $this->unitsPerEm = $this->readUInt(2);
  248.  248         $this->_debugLog('Units per em: %d', $this->unitsPerEm);
  249.  249 
  250.  250         /* Skip creation and modification date/time.
  251.  251          */
  252.  252         $this->skipBytes(16);
  253.  253 
  254.  254         $this->xMin = $this->readInt(2);
  255.  255         $this->yMin = $this->readInt(2);
  256.  256         $this->xMax = $this->readInt(2);
  257.  257         $this->yMax = $this->readInt(2);
  258.  258         $this->_debugLog('Font bounding box: %d %d %d %d',
  259.  259                          $this->xMin, $this->yMin, $this->xMax, $this->yMax);
  260.  260 
  261.  261         /* The style bits here must match the fsSelection bits in the OS/2
  262.  262          * table, if present.
  263.  263          */
  264.  264         $macStyleBits = $this->readUInt(2);
  265.  265         $this->isBold   = $this->isBitSet(0, $macStyleBits);
  266.  266         $this->isItalic = $this->isBitSet(1, $macStyleBits);
  267.  267 
  268.  268         /* We don't need the remainder of data in this table: smallest readable
  269.  269          * size, font direction hint, indexToLocFormat, and glyphDataFormat.
  270.  270          */
  271.  271     }
  272.  272 
  273.  273 
  274.  274     /**
  275.  275      * Parses the OpenType name (Naming) table.
  276.  276      *
  277.  277      * The name table contains all of the identifying strings associated with
  278.  278      * the font such as its name, copyright, trademark, license, etc.
  279.  279      *
  280.  280      * @throws Zend_Pdf_Exception
  281.  281      */
  282.  282     protected function _parseNameTable()
  283.  283     {
  284.  284         $this->_jumpToTable('name');
  285.  285         $baseOffset = $this->_tableDirectory['name']['offset'];
  286.  286 
  287.  287         /* The name table begins with a short header, followed by each of the
  288.  288          * fixed-length name records, followed by the variable-length strings.
  289.  289          */
  290.  290 
  291.  291         /* We only understand version 0 tables.
  292.  292          */
  293.  293         $tableFormat = $this->readUInt(2);
  294.  294         if ($tableFormat != 0) {
  295.  295             require_once 'Zend/Pdf/Exception.php';
  296.  296             throw new Zend_Pdf_Exception("Unable to read format $tableFormat table",
  297.  297                                          Zend_Pdf_Exception::DONT_UNDERSTAND_TABLE_VERSION);
  298.  298         }
  299.  299         $this->_debugLog('Format %d table', $tableFormat);
  300.  300 
  301.  301         $nameCount = $this->readUInt(2);
  302.  302         $this->_debugLog('%d name strings', $nameCount);
  303.  303 
  304.  304         $storageOffset = $this->readUInt(2) + $baseOffset;
  305.  305         $this->_debugLog('Storage offset: 0x%x', $storageOffset);
  306.  306 
  307.  307         /* Scan the name records for those we're interested in. We'll skip over
  308.  308          * encodings and languages we don't understand or support. Prefer the
  309.  309          * Microsoft Unicode encoding for a given name/language combination, but
  310.  310          * use Mac Roman if nothing else is available. We will extract the
  311.  311          * actual strings later.
  312.  312          */
  313.  313         $nameRecords = array();
  314.  314         for ($nameIndex = 0; $nameIndex < $nameCount; $nameIndex++) {
  315.  315 
  316.  316             $platformID = $this->readUInt(2);
  317.  317             $encodingID = $this->readUInt(2);
  318.  318 
  319.  319             if (! ( (($platformID == 3) && ($encodingID == 1)) ||    // Microsoft Unicode
  320.  320                     (($platformID == 1) && ($encodingID == 0))       // Mac Roman
  321.  321                    ) ) {
  322.  322                 $this->skipBytes(8);    // Not a supported encoding. Move on.
  323.  323                 continue;
  324.  324             }
  325.  325 
  326.  326             $languageID = $this->readUInt(2);
  327.  327             $nameID     = $this->readUInt(2);
  328.  328             $nameLength = $this->readUInt(2);
  329.  329             $nameOffset = $this->readUInt(2);
  330.  330 
  331.  331             $languageCode = $this->_languageCodeForPlatform($platformID, $languageID);
  332.  332             if ($languageCode === null) {
  333.  333                 $this->_debugLog('Skipping languageID: 0x%x; platformID %d', $languageID, $platformID);
  334.  334                 continue;    // Not a supported language. Move on.
  335.  335             }
  336.  336 
  337.  337             $this->_debugLog('Adding nameID: %d; languageID: 0x%x; platformID: %d; offset: 0x%x (0x%x); length: %d',
  338.  338                              $nameID, $languageID, $platformID, $baseOffset + $nameOffset, $nameOffset, $nameLength);
  339.  339 
  340.  340             /* Entries in the name table are sorted by platform ID. If an entry
  341.  341              * exists for both Mac Roman and Microsoft Unicode, the Unicode entry
  342.  342              * will prevail since it is processed last.
  343.  343              */
  344.  344             $nameRecords[$nameID][$languageCode] = array('platform' => $platformID,
  345.  345                                                          'offset'   => $nameOffset,
  346.  346                                                          'length'   => $nameLength );
  347.  347         }
  348.  348 
  349.  349         /* Now go back and extract the interesting strings.
  350.  350          */
  351.  351         $fontNames = array();
  352.  352         foreach ($nameRecords as $name => $languages) {
  353.  353             foreach ($languages as $language => $attributes) {
  354.  354                 $stringOffset = $storageOffset + $attributes['offset'];
  355.  355                 $this->moveToOffset($stringOffset);
  356.  356                 if ($attributes['platform'] == 3) {
  357.  357                     $string = $this->readStringUTF16($attributes['length']);
  358.  358                 } else {
  359.  359                     $string = $this->readStringMacRoman($attributes['length']);
  360.  360                 }
  361.  361                 $fontNames[$name][$language] = $string;
  362.  362             }
  363.  363         }
  364.  364 
  365.  365         $this->names = $fontNames;
  366.  366     }
  367.  367 
  368.  368 
  369.  369     /**
  370.  370      * Parses the OpenType post (PostScript Information) table.
  371.  371      *
  372.  372      * The post table contains additional information required for using the font
  373.  373      * on PostScript printers. It also contains the preferred location and
  374.  374      * thickness for an underline, which is used by our layout code.
  375.  375      *
  376.  376      * @throws Zend_Pdf_Exception
  377.  377      */
  378.  378     protected function _parsePostTable()
  379.  379     {
  380.  380         $this->_jumpToTable('post');

  493.  493     protected function _parseOs2Table()
  494.  494     {

  517.  517                 $this->isOrnamentalFont = false;
  518.  518                 $this->isScriptFont     = false;

  530.  530             }
  531.  531         }
  532.  532 
  533.  533         /* Version 0 tables are becoming rarer these days. They are only found
  534.  534          * in older fonts.
  535.  535          *
  536.  536          * Version 1 formally defines the Unicode character range bits and adds
  537.  537          * two new fields to the end of the table.
  538.  538          *
  539.  539          * Version 2 defines several additional flags to the embedding bits

  541.  541          *
  542.  542          * Versions 2 and 3 are structurally identical. There are only two
  543.  543          * significant differences between the two: First, in version 3, the

  545.  545          * non-zero width glyphs in the font instead of just the Latin lower-
  546.  546          * case alphabetic characters; this doesn't affect us. Second, in
  547.  547          * version 3, the embedding bits (fsType field) have been made mutually
  548.  548          * exclusive; see additional discusson on this below.
  549.  549          *
  550.  550          * We can understand all four of these table versions.

  553.  553         if (($tableVersion < 0) || ($tableVersion > 3)) {
  554.  554             require_once 'Zend/Pdf/Exception.php';

  556.  556                                          Zend_Pdf_Exception::DONT_UNDERSTAND_TABLE_VERSION);
  557.  557         }

  559.  559 
  560.  560         $this->averageCharWidth = $this->readInt(2);
  561.  561 
  562.  562         /* Indicates the visual weight and aspect ratio of the characters. Used
  563.  563          * primarily to logically sort fonts in lists. Also used to help choose
  564.  564          * a more appropriate substitute font when necessary. See the WEIGHT_
  565.  565          * and WIDTH_ constants defined in Zend_Pdf_Font.
  566.  566          */
  567.  567         $this->fontWeight = $this->readUInt(2);
  568.  568         $this->fontWidth  = $this->readUInt(2);
  569.  569 
  570.  570         /* Describes the font embedding licensing rights. We can only embed and
  571.  571          * subset a font when given explicit permission.
  572.  572          *
  573.  573          * NOTE: We always interpret these bits according to the rules defined
  574.  574          * in version 3 of this table, regardless of the actual version. This
  575.  575          * means we will perform our checks in order from the most-restrictive
  576.  576          * to the least.
  577.  577          */

  582.  582              * outlines from fonts yet, so this means no embed.

  585.  585         } else if ($this->isBitSet(1, $embeddingFlags)) {

  587.  587              * enforce this, so interpret this as no embed. This may be revised
  588.  588              * in the future...
  589.  589              */
  590.  590             $this->isEmbeddable = false;
  591.  591         } else {
  592.  592             /* The remainder of the bit settings grant us permission to embed
  593.  593              * the font. There may be additional usage rights granted or denied
  594.  594              * but those only affect the PDF viewer application, not our code.

  596.  596             $this->isEmbeddable = true;
  597.  597         }
  598.  598         $this->_debugLog('Font ' . ($this->isEmbeddable ? 'may' : 'may not') . ' be embedded');

  600.  600 
  601.  601         /* Recommended size and offset for synthesized subscript characters.

  603.  603         $this->subscriptXSize = $this->readInt(2);
  604.  604         $this->subscriptYSize = $this->readInt(2);
  605.  605         $this->subscriptXOffset = $this->readInt(2);
  606.  606         $this->subscriptYOffset = $this->readInt(2);
  607.  607 
  608.  608         /* Recommended size and offset for synthesized superscript characters.

  611.  611         $this->superscriptYSize = $this->readInt(2);

  614.  614

  622.  622          *   http://www.microsoft.com/OpenType/OTSpec/ibmfc.htm

  671.  671          */
  672.  672         $this->skipBytes(6);
  673.  673 
  674.  674         /* Typographic ascender, descender, and line gap. These values are
  675.  675          * preferred to those in the 'hhea' table.
  676.  676          */
  677.  677         $this->ascent = $this->readInt(2);
  678.  678         $this->descent = $this->readInt(2);
  679.  679         $this->lineGap = $this->readInt(2);
  680.  680 
  681.  681         /* The descent value is supposed to be negative--it's the distance
  682.  682          * relative to the baseline. However, some fonts improperly store a
  683.  683          * positive value in this field. If a positive value is found, flip the
  684.  684          * sign and record a warning in the debug log that we did this.
  685.  685          */
  686.  686         if ($this->descent > 0) {
  687.  687             $this->_debugLog('Warning: Font should specify negative descent. Actual: %d; Using %d',
  688.  688                              $this->descent, -$this->descent);
  689.  689             $this->descent = -$this->descent;
  690.  690         }
  691.  691 
  692.  692         /* Skip over Windows-specific ascent and descent.
  693.  693          */
  694.  694         $this->skipBytes(4);
  695.  695 
  696.  696         /* Versions 0 and 1 tables do not contain the x or capital height
  697.  697          * fields. Record zero for unknown.
  698.  698          */

  700.  700             $this->xHeight = 0;
  701.  701             $this->capitalHeight = 0;
  702.  702         } else {
  703.  703 
  704.  704             /* Skip over the Windows code page coverages. We are only concerned
  705.  705              * with Unicode coverage.
  706.  706              */
  707.  707             $this->skipBytes(8);
  708.  708 
  709.  709             $this->xHeight = $this->readInt(2);
  710.  710             $this->capitalHeight = $this->readInt(2);
  711.  711
