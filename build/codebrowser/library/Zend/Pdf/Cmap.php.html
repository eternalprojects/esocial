  259. 259     /**
  260. 260      * Extracts a signed 2-byte integer from a string.
  261. 261      *
  262. 262      * Integers are always big-endian. Throws an exception if the index is out
  263. 263      * of range.

  271. 271     {

  273. 273             require_once 'Zend/Pdf/Exception.php';
  274. 274             throw new Zend_Pdf_Exception("Index out of range: $index",
  275. 275                                          Zend_Pdf_Exception::INDEX_OUT_OF_RANGE);
  276. 276         }
  277. 277         $number = ord($data[$index]);
  278. 278         if (($number & 0x80) == 0x80) {    // negative
  279. 279             $number = ~((((~ $number) & 0xff) << 8) | ((~ ord($data[++$index])) & 0xff));

  282. 282         }

  284. 284     }
