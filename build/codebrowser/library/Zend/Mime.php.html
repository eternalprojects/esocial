  187. 187         $lineLength = self::LINELENGTH,
  188. 188         $lineEnd = self::LINEEND)
  189. 189     {

  191. 191         $prefix = sprintf('=?%s?Q?', $charset);
  192. 192         $lineLength = $lineLength-strlen($prefix)-3;

  195. 195 
  196. 196         // Mail-Header required chars have to be encoded also:

  198. 198 
  199. 199         // initialize first line, we need it anyways

  201. 201 
  202. 202         // Split encoded text into separate lines
  203. 203         $tmp = "";

  205. 205             $currentLine = max(count($lines)-1, 0);
  206. 206             $token       = self::getNextQuotedPrintableToken($str);
  207. 207             $str         = substr($str, strlen($token));
  208. 208 
  209. 209             $tmp .= $token;

  211. 211                 // only if we have a single char token or space, we can append the
  212. 212                 // tempstring it to the current line or start a new line if necessary.
  213. 213                 if(strlen($lines[$currentLine].$tmp) > $lineLength) {
  214. 214                     $lines[$currentLine+1] = $tmp;
  215. 215                 } else {
  216. 216                     $lines[$currentLine] .= $tmp;
  217. 217                 }
  218. 218                 $tmp = "";
  219. 219             }
  220. 220             // don't forget to append the rest to the last line
  221. 221             if(strlen($str) == 0) {
  222. 222                 $lines[$currentLine] .= $tmp;
  223. 223             }
  224. 224         }
  225. 225 
  226. 226         // assemble the lines together by pre- and appending delimiters, charset, encoding.
  227. 227         for($i = 0; $i < count($lines); $i++) {
