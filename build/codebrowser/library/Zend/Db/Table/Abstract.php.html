  790.  790     {

  792.  792             return true;
  793.  793         }

  802.  802         }

  805.  805         if (null !== $this->_metadataCache) {

  807.  807

  814.  814                   ? ':'.$dbConfig['port']
  815.  815                   : null);
  816.  816 
  817.  817             $host = isset($dbConfig['options']['host'])
  818.  818                   ? ':'.$dbConfig['options']['host']

  820.  820                   ? ':'.$dbConfig['host']

  826.  826                   . $this->_schema. '.' . $this->_name
  827.  827             );
  828.  828         }

  839.  839             }
  840.  840         }
  841.  841 
  842.  842         // Assign the metadata to $this
  843.  843         $this->_metadata = $metadata;
  844.  844 
  845.  845         // Return whether the metadata were loaded from cache
  846.  846         return $isMetadataFromCache;
  847.  847     }
